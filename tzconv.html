Here’s the PowerShell version of the timezone conversion function with the same features as the Fish version, including the new -TimeQuery parameter:

function Convert-TimeZone {
    [CmdletBinding()]
    param(
        [Parameter(Mandatory=$false)]
        [Alias('f')]
        [string]$From,

        [Parameter(Mandatory=$true)]
        [Alias('t')]
        [string]$To,

        [Parameter(Mandatory=$false)]
        [Alias('d')]
        [switch]$Dst,

        [Parameter(Mandatory=$false)]
        [Alias('F')]
        [string]$Format = "HH:mm",

        [Parameter(Mandatory=$false)]
        [Alias('o')]
        [string]$Offset,

        [Parameter(Mandatory=$false)]
        [Alias('q')]
        [string]$TimeQuery,

        [Parameter(Mandatory=$false)]
        [Alias('h')]
        [switch]$Help
    )

    # Timezone mapping dictionary
    $timezoneMap = @{
        'est'  = 'Eastern Standard Time'
        'edt'  = 'Eastern Standard Time'
        'cst'  = 'Central Standard Time'
        'cdt'  = 'Central Standard Time'
        'mst'  = 'Mountain Standard Time'
        'mdt'  = 'Mountain Standard Time'
        'pst'  = 'Pacific Standard Time'
        'pdt'  = 'Pacific Standard Time'
        'irst' = 'Iran Standard Time'
        'irdt' = 'Iran Standard Time'
        'utc'  = 'UTC'
        'gmt'  = 'GMT Standard Time'
        'jst'  = 'Tokyo Standard Time'
        'cet'  = 'Central European Standard Time'
        'cest' = 'Central European Standard Time'
        'aedt' = 'AUS Eastern Standard Time'
        'aest' = 'AUS Eastern Standard Time'
        'nzst' = 'New Zealand Standard Time'
        'nzdt' = 'New Zealand Standard Time'
        'ist'  = 'India Standard Time'
    }

    # Show help if requested
    if ($Help) {
        Write-Host "Usage: Convert-TimeZone [OPTIONS]"
        Write-Host "Convert time between time zones"
        Write-Host ""
        Write-Host "Options:"
        Write-Host "  -f, --from TIMEZONE    Source timezone (default: localtime when using -q)"
        Write-Host "  -t, --to TIMEZONE      Target timezone (required)"
        Write-Host "  -d, --dst              Use DST for offset (off by default)"
        Write-Host "  -F, --format FORMAT    Output format (.NET format, default: 'HH:mm')"
        Write-Host "  -o, --offset TIMEZONE  Print UTC offset for TIMEZONE (e.g., '+3:30')"
        Write-Host "  -q, --timequery TIME   Time to convert (24-hour format, e.g., '02:33')"
        Write-Host "  -h, --help             Show this help"
        Write-Host ""
        Write-Host "Supported timezone abbreviations:"
        Write-Host ($timezoneMap.Keys | Sort-Object | ForEach-Object { $_.ToUpper() }) -Separator ', '
        return
    }

    # Handle offset request
    if ($Offset) {
        $tzKey = $Offset.ToLower()
        if (-not $timezoneMap.ContainsKey($tzKey)) {
            Write-Host "Unsupported timezone: $Offset" -ForegroundColor Red
            Write-Host "Supported timezones: $($timezoneMap.Keys | Sort-Object | ForEach-Object { $_.ToUpper() })" -Separator ', '
            return
        }
        
        $tzId = $timezoneMap[$tzKey]
        $tz = [System.TimeZoneInfo]::FindSystemTimeZoneById($tzId)
        
        # Choose date for DST/non-DST calculation
        $date = if ($Dst) { [DateTime]::Now } else { [DateTime]::Parse('2023-01-01') }
        
        $utcTime = [DateTime]::SpecifyKind($date, [DateTimeKind]::Utc)
        $offsetSpan = $tz.GetUtcOffset($utcTime)
        
        $sign = if ($offsetSpan.TotalMinutes -ge 0) { '+' } else { '-' }
        $hours = [Math]::Abs($offsetSpan.Hours)
        $minutes = [Math]::Abs($offsetSpan.Minutes)
        
        return "$sign$hours`:$minutes"
    }

    # Validate target timezone
    $toKey = $To.ToLower()
    if (-not $timezoneMap.ContainsKey($toKey)) {
        Write-Host "Unsupported target timezone: $To" -ForegroundColor Red
        Write-Host "Supported timezones: $($timezoneMap.Keys | Sort-Object | ForEach-Object { $_.ToUpper() })" -Separator ', '
        return
    }
    $toTz = [System.TimeZoneInfo]::FindSystemTimeZoneById($timezoneMap[$toKey])

    # Handle time conversion
    if ($TimeQuery) {
        # Parse explicit time query
        try {
            $time = [DateTime]::ParseExact($TimeQuery, "HH:mm", $null)
        }
        catch {
            Write-Host "Invalid time format: $TimeQuery. Please use 24-hour format (e.g., '15:30')" -ForegroundColor Red
            return
        }

        # Get source timezone
        $fromTz = if ($From) {
            $fromKey = $From.ToLower()
            if (-not $timezoneMap.ContainsKey($fromKey)) {
                Write-Host "Unsupported source timezone: $From" -ForegroundColor Red
                Write-Host "Supported timezones: $($timezoneMap.Keys | Sort-Object | ForEach-Object { $_.ToUpper() })" -Separator ', '
                return
            }
            [System.TimeZoneInfo]::FindSystemTimeZoneById($timezoneMap[$fromKey])
        } else {
            [System.TimeZoneInfo]::Local
        }

        # Create DateTimeOffset with source timezone
        $sourceDateTime = [DateTime]::SpecifyKind(
            (Get-Date -Date $time -Hour $time.Hour -Minute $time.Minute -Second 0), 
            [DateTimeKind]::Unspecified
        )
        $sourceOffset = [DateTimeOffset]::new($sourceDateTime, $fromTz.GetUtcOffset($sourceDateTime))

        # Convert to target timezone
        $targetTime = [TimeZoneInfo]::ConvertTime($sourceOffset, $toTz)
        return $targetTime.ToString($Format)
    }
    else {
        # Convert current time to target timezone
        $utcNow = [DateTime]::UtcNow
        $targetTime = [TimeZoneInfo]::ConvertTimeFromUtc($utcNow, $toTz)
        return $targetTime.ToString($Format)
    }
}

Key Features:

  1. Timezone Mapping:
    • Uses .NET’s TimeZoneInfo system for accurate conversions
  • Case-insensitive timezone abbreviations (e.g., est, PST)
  • Supports all major timezones with DST awareness
  1. New -TimeQuery Parameter:
Convert-TimeZone -To est -TimeQuery "15:30"
  • Converts 3:30 PM (local time) to EST
  • Combine with -From to specify source timezone:
Convert-TimeZone -From utc -To pst -TimeQuery "12:00"
  1. Offset Calculation:
Convert-TimeZone -Offset irst -Dst
# Returns +3:30 (with DST)
  1. Custom Formatting:
Convert-TimeZone -To cet -Format "yyyy-MM-dd HH:mm"
  1. Help Documentation:
Convert-TimeZone -Help

Usage Examples:

  1. Convert current time to another timezone:
Convert-TimeZone -To ist
# Output: 21:45 (example)
  1. Convert specific time between timezones:
Convert-TimeZone -From pst -To gmt -TimeQuery "09:30"
# Converts 9:30 AM PST to GMT
  1. Get timezone offset:
Convert-TimeZone -Offset aest
# Returns +10:00 (standard offset)
  1. DST-aware offset:
Convert-TimeZone -Offset edt -Dst
# Returns -4:00 (during DST period)
  1. Custom date/time formatting:
Convert-TimeZone -To jst -TimeQuery "23:15" -Format "dddd, MMMM dd @ HH:mm"
# Output: "Friday, October 06 @ 08:15"

Notes:

  1. The function uses .NET’s built-in timezone database for accurate conversions
  2. Timezone abbreviations are case-insensitive (e.g., est, EST, eSt all work)
  3. When using -TimeQuery without -From, the local system timezone is used as source
  4. The -Dst switch only affects offset calculations (-Offset parameter)
  5. Format strings use .NET’s datetime format specifiers (e.g., HH:mm for 24-hour time)

To use this function, simply add it to your PowerShell profile or run it directly in your session. The function provides comprehensive error handling for invalid inputs and unsupported timezones.
